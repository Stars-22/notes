## 力扣：739. 每日温度
**链接**：https://leetcode.cn/problems/daily-temperatures/  
**相关标签**：`栈` `数组` `单调栈`  
**难度**：`中等`
### 方法1：`单调栈`
- **时间复杂度**：$O(n)$
- **空间复杂度**：$O(n)$
```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& t) {
        vector<int> ans(t.size(), 0);
        stack<int> st;
        for(int i=t.size()-1; i>=0; i--){
            while(!st.empty() && t[i] >= t[st.top()]) st.pop();
            if(!st.empty()) ans[i] = st.top() - i;
            st.push(i);
        }
        return ans;
    }
};
```

## 力扣：322. 零钱兑换
**链接**：https://leetcode.cn/problems/coin-change/  
**相关标签**：`广度优先搜索` `数组` `动态规划`  
**难度**：`中等`
### 方法1：`动态规划`
- **时间复杂度**：$O(Sn)$
- **空间复杂度**：$O(S)$
```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int a[10001] = {0};
        memset(a+1, 1, 10000*sizeof(int));
        for(const auto& m : coins)
            for(int j=m; j<=amount; j++)
                a[j] = min(a[j], a[j-m]+1);
        if(a[amount] == 16843009) return -1;
        return a[amount];
    }
};
```
