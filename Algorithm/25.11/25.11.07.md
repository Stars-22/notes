## 力扣：438. 找到字符串中所有字母异位词
**链接**：https://leetcode.cn/problems/find-all-anagrams-in-a-string/  
**相关标签**：`哈希表` `字符串` `滑动窗口`  
**难度**：`中等`
### 方法1：`双指针（滑动窗口）`
- **时间复杂度**：$O(nΣ)$
- **空间复杂度**：$O(Σ)$
```cpp
class Solution {
public:
    unordered_map<char, short> k;
    bool isP(){
        for(const auto& k : k) if(k.second != 0) return false;
        return true;
    }
    vector<int> findAnagrams(string s, string p) {
        vector<int> ans;
        if(s.size() < p.size()) return ans;
        for(const auto& p : p) k[p]++;
        int l = 0, r = 0;
        while(r < s.size()){
            if(k.find(s[r]) != k.end()){
                k[s[r]] --;
                if(r - l + 1 == p.size()){
                    if(isP()) ans.push_back(l);
                    k[s[l++]]++;
                }
            }
            else{
                while(l < r) k[s[l++]]++;
                l++;
            }
            r++;
        }
        return ans;
    }
};
```
### 方法2：`滑动窗口`
- **时间复杂度**：$O(nΣ)$
- **空间复杂度**：$O(Σ)$
```cpp
class Solution {
public:
    int a[8], n;
    vector<string> ans;
    void creatAns(){
        string s;
        for(int i=0; i<n; i++) s += '(' + string(a[i], ')');
        ans.push_back(move(s));
    }
    void dfs(int k, int n){
       for(int i=k<n?1:0; i<=n; i++){
           a[k] = i;
           if(i == n) creatAns();
           else dfs(k-1, n-i);
       }
       a[k] = 0;
    }
    vector<string> generateParenthesis(int n) {
        this->n = n;
        dfs(n-1, n);
        return ans;
    }
};
```

## 力扣：543. 二叉树的直径
**链接**：https://leetcode.cn/problems/diameter-of-binary-tree/  
**相关标签**：`树` `深度优先搜索` `二叉树`  
**难度**：`简单`
### 方法1：`深度优先搜索`
- **时间复杂度**：$O(n)$
- **空间复杂度**：$O(h)$
```cpp
class Solution {
public:
    int ans = 0;
    int dfs(TreeNode* root){
        if(!root) return 0;
        int l = dfs(root->left);
        int r = dfs(root->right);
        ans = max(ans, l+r);
        return max(l, r) + 1;
    }
    int diameterOfBinaryTree(TreeNode* root) {
        dfs(root);
        return ans;
    }
};
```
