## 力扣：148. 排序链表
**链接**：https://leetcode.cn/problems/sort-list/  
**相关标签**：`链表` `双指针` `分治` `排序` `归并排序`  
**难度**：`中等`
### 方法1：`归并排序`
- **时间复杂度**：$O(n \log n)$
- **空间复杂度**：$O(1)$
```cpp
class Solution {
private:
    ListNode* mergeList(ListNode* a, ListNode* b){
        ListNode dummy;
        ListNode* cur = &dummy;
        while(a && b){
            if(a->val < b->val){
                cur->next = a;
                a = a->next;
            }
            else{
                cur->next = b;
                b = b->next;
            }
            cur = cur->next;
        }
        cur->next = a ? a : b;
        return dummy.next;
    }
    ListNode* cutList(ListNode* head, int n){
        if (!head) return nullptr;
        for(int i=1; i<n && head->next; i++) head = head->next;
        ListNode* ans = head->next;
        head->next = nullptr;
        return ans;
    }
public:
    ListNode* sortList(ListNode* head) {
        int l = 0;
        ListNode dummy(0, head);
        ListNode* cur = head;
        while(cur){
            cur = cur->next;
            l++;
        }
        for(int i=1; i<l; i*=2){
            ListNode* tail = &dummy;
            cur = dummy.next;
            while(cur){
                ListNode* head1 = cur;
                ListNode* head2 = cutList(cur, i);
                cur = cutList(head2, i);
                tail->next = mergeList(head1, head2);
                while(tail->next) tail = tail->next;
            }   
        }
        return dummy.next;
    }
};
```
