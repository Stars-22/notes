## 力扣：3606. 优惠券校验器
**链接**：https://leetcode.cn/problems/coupon-code-validator/  
**相关标签**：`数组` `哈希表` `字符串` `排序`  
**难度**：`简单`
### 方法1：`模拟`
- **时间复杂度**：$O(nL \log n)$
- **空间复杂度**：$O(nL)$
```cpp
class Solution {
    bool isValid(const string& s){
        for(char c : code[i]) if(!(isalnum(c) || c == '_')) return false;
        return !s.empty();
    }
public:
    vector<string> validateCoupons(vector<string>& code, vector<string>& businessLine, vector<bool>& isActive) {
        static const unordered_map<string, int> businessPriority = {
            {"electronics", 0}, {"grocery", 1}, {"pharmacy", 2}, {"restaurant", 3}
        };
        vector<pair<int, string>> validCoupons;
        for(int i=0; i<code.size(); i++){
            if(!isActive[i] || !isValid(code[i])) continue;
            if(businessPriority.find(businessLine[i]) == businessPriority.end()) continue;
            validCoupons.emplace_back(businessPriority.at(businessLine[i]), code[i]);
        }
        sort(validCoupons.begin(), validCoupons.end());
        vector<string> result;
        for(const auto& [_, couponCode] : validCoupons) result.push_back(couponCode);
        return result;
    }
};
```
