## 洛谷：P2142 高精度减法
**链接**：https://www.luogu.com.cn/problem/P2142/  
**相关标签**：`模拟` `字符串` `高精度` `洛谷原创`  
**难度**：`普及-`
### 方法1：`模拟`
- **时间复杂度**：$O(n)$
- **空间复杂度**：$O(n)$
```cpp
#include "bits/stdc++.h"
using namespace std;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    char a[10090], b[10090], ans[10090];
    bool minus = false;
    cin >> a;
    cin >> b;
    if (strlen(a) < strlen(b)) {
        swap(a, b);
        minus = true;
    }
    else if(strlen(a) == strlen(b)){
        for(int i = 0; a[i]; i++){
            if(a[i] < b[i]){
                swap(a, b);
                minus = true;
                break;
            }
            else if(a[i] > b[i]) break;
        }
    }
    int n = strlen(a);
    int d = n - strlen(b);
    memset(ans, '0',  n);
    for(int i = n - 1; i >= 0; i--) {
        ans[i] += a[i] - '0' - (i - d >= 0 ? b[i - d] - '0' : 0);
        if(ans[i] < '0'){
            ans[i] += 10;
            ans[i-1] = '0' - 1;
        }
    }
    bool k = true;
    if(minus) cout << '-';
    for(int i=0;i<n;i++) {
        if(ans[i] == '0' && k && i!=n-1) continue;
        k = false;
        cout << ans[i];
    }
    cout << endl;
    return 0;
}
```
