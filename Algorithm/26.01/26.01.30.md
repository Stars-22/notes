## 洛谷：P1009 [NOIP 1998 普及组] 阶乘之和
**链接**：https://www.luogu.com.cn/problem/P1009/  
**相关标签**：`数学` `高精度` `15'未知'` `82'未知'`  
**难度**：`普及-`
### 方法1：``
- **时间复杂度**：$O(mn)$
- **空间复杂度**：$O(n)$
```cpp
#include "bits/stdc++.h"
#include <cstring>
using namespace std;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n, l_t = 1;
    cin >> n;
    int temp[102], ans[102];
    memset(temp, 0, sizeof(temp));
    memset(ans, 0, sizeof(ans));
    temp[0] = 1;
    ans[0] = 1;
    for(int i=2; i<=n; i++){
        int add = 0;
        for(int j=0; j<l_t; j++){
            temp[j] = temp[j] * i + add;
            add = temp[j] / 10000;
            temp[j] %= 10000;
            ans[j] += temp[j];
            ans[j+1] += ans[j] / 10000;
            ans[j] %= 10000;
            if(add)  l_t = max(l_t, j+2);
        }
    }
    for(int i=100; i>=0; --i){
        if(i == 0 || ans[i] != 0){
            cout << ans[i--];
            for(; i>=0; --i){
                if(ans[i] < 1000) cout << "0";
                if(ans[i] < 100) cout << "0";
                if(ans[i] < 10) cout << "0";
                cout << ans[i];
            }
        }
    }
    return 0;
}
```
