## 洛谷：P1480 A/B Problem（高精度除法Ⅰ）
**链接**：https://www.luogu.com.cn/problem/P1480/  
**相关标签**：`模拟` `高精度`  
**难度**：`普及-`
### 方法1：`模拟`
- **时间复杂度**：$O(n)$
- **空间复杂度**：$O(n)$
```cpp
#include "bits/stdc++.h"
using namespace std;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    char a[5002], ans[5002];
    long long b;
    cin >> a >> b;
    long long n = strlen(a), temp = 0;
    memset(ans, 0, sizeof(ans));
    for(int i=0; i<n; ++i) a[i] = a[i] - '0';
    for(int i=0; i<n; ++i){
        temp *= 10;
        temp += a[i];
        ans[i] = temp / b;
        temp %= b;
    }
    for(int i=0; i<n; ++i)
        if(ans[i] != 0 || i == n-1)
            for(; i<n; ++i) cout << (int)ans[i];
    return 0;
}
```
